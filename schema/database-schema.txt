CREATE DATABASE OAUTHKEEPER;
USE DATABASE OAUTHKEEPER;


CREATE TABLE USERS (
	USERID INT(10) AUTO_INCREMENT PRIMARY KEY,
	NAME VARCHAR(30) NOT NULL,

	ADMIN BIT NOT NULL,

	USERNAME VARCHAR(30) NOT NULL UNIQUE,
	PASSWORD VARCHAR(200) NOT NULL
)


CREATE TABLE CONTACTS (
	USERID INT(10) AUTO_INCREMENT PRIMARY KEY,
	NAME VARCHAR(80) NOT NULL,
	COMPANY VARCHAR(80) NOT NULL,
	MOBILE VARCHAR(80) NOT NULL UNIQUE,
	EMAIL VARCHAR(80) NOT NULL UNIQUE,
	ADDRESS VARCHAR(500),
	STATUS ENUM ("NOT CALLED", "WRONG NUMBER", "CALLED/DECLINED", "CALLED/NOT REACHABLE",  "CALLED/POSTPONED", "CALLED/ACCEPTED", "EMAILED/AWAITING RESPONSE", "EMAILED/DECLINED", "EMAILED/CONFIRMED") DEFAULT "NOT CALLED"
);
